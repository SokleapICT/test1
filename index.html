⚡ ការណែនាំអំពីការចងភ្ជាប់ជាមួយ Supabase

កម្មវិធីនេះត្រូវបានរៀបចំឡើងយ៉ាងស្អាត និងមានរចនាសម្ព័ន្ធទិន្នន័យ (Data Object) ច្បាស់លាស់។ លោកអ្នកអាចភ្ជាប់កម្មវិធីនេះទៅកាន់ Supabase Database របស់អ្នកបានយ៉ាងងាយស្រួលបំផុតដោយគ្រាន់តែអនុវត្តតាមជំហានសាមញ្ញៗខាងក្រោម៖

ជំហានទី១៖ បន្ថែម Supabase Library ទៅក្នុង Header <head> <script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2"></script>
ជំហានទី២៖ បង្កើត Client Supabase ក្នុង Javascript 
const supabaseUrl = 'YOUR_SUPABASE_PROJECT_URL';
const supabaseKey = 'YOUR_SUPABASE_ANON_KEY';
const supabase = supabase.createClient(supabaseUrl, supabaseKey);
ជំហានទី៣៖ កូដសម្រាប់បញ្ចូលទិន្នន័យសិស្សទៅកាន់ Database

(ជំនួសមុខងាររក្សាទុកដែលមានស្រាប់ទៅជាមុខងារ Supabase ដូចខាងក្រោម៖)

async function saveToSupabase(newStudent) {
    const { data, error } = await supabase
        .from('students') // matching your table name
        .insert([newStudent]);

    if (error) {
        alert('មានបញ្ហាក្នុងការរក្សាទុក៖ ' + error.message);
    } else {
        alert('ទិន្នន័យត្រូវបានរក្សាទុកទៅកាន់ Supabase ដោយជោគជ័យ!');
    }
}
